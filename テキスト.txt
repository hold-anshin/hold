  document.getElementById(id).classList.remove('hidden');
}

function switchTab(t) {
  if (t === 'chat') {
    showPage('p4');
  } else {
    showPage('p5');
  }
}

function getTime() {
  return new Date().toLocaleTimeString('ja-JP',{hour:'2-digit',minute:'2-digit'});
}

function addMsg(text, type, isClosing) {
  var area = document.getElementById('chat-body');
  var div = document.createElement('div');
  div.className = 'msg ' + type + (isClosing ? ' closing' : '');
  div.innerHTML = '<div class="bubble">' + text.replace(/\n/g,'<br>') + '</div><div class="mtime">' + (type==='bot'?'HOLD':getTime()) + '</div>';
  area.appendChild(div);
  area.scrollTop = area.scrollHeight;
  saveHistory();
}

function showTyping() {
  var area = document.getElementById('chat-body');
  var div = document.createElement('div');
  div.className = 'msg bot'; div.id = 'typing';
  div.innerHTML = '<div class="typing"><span></span><span></span><span></span></div>';
  area.appendChild(div); area.scrollTop = area.scrollHeight;
}
function removeTyping() { var t=document.getElementById('typing'); if(t)t.remove(); }

function getReply() {
  if (phase === 'initial') { phase = 'deep1'; return DEEP_FLOW[0]; }
  if (phase === 'deep1')   { phase = 'deep2'; return DEEP_FLOW[1]; }
  if (phase === 'deep2')   { phase = 'deep3'; return DEEP_FLOW[2]; }
  if (phase === 'deep3')   { phase = 'deep4'; return DEEP_FLOW[3]; }
  if (phase === 'deep4')   { phase = 'closing'; return DEEP_FLOW[4]; }
  if (phase === 'closing') { phase = 'done'; return null; }
  return null;
}

function handleSend(text) {
  if (!text) return;
  addMsg(text, 'user');
  var btn = document.getElementById('send-btn');
  btn.disabled = true;
  showTyping();
  setTimeout(function() {
    removeTyping();
    var reply = getReply();
    if (!reply) {
      // closing後の返信
      var area = document.getElementById('chat-body');
      var closing = 'それではあせらず、少しずつやってみませんか？\n\nよかったら、心がほっとするような\n小さな行動を見てください。\n\nまずは自分を安心させてあげることが\n一番の近道だから。🌿';
      var div = document.createElement('div');
      div.className = 'msg bot closing';
      div.innerHTML = '<div class="bubble">' + closing.replace(/\n/g,'<br>') + '</div><div class="mtime">HOLD</div>';
      area.appendChild(div);
      area.scrollTop = area.scrollHeight;
      saveHistory();
      setTimeout(function(){
        var gb = document.createElement('button');
        gb.className = 'btn-go';
        gb.textContent = '🌿 小さな安心行動を見てみる';
        gb.onclick = function(){ switchTab('cards'); };
        area.appendChild(gb);
        var rb = document.createElement('button');
        rb.className = 'btn-restart';
        rb.textContent = '🌿 新しい気持ちを話す';
        rb.onclick = restartChat;
        area.appendChild(rb);
        area.scrollTop = area.scrollHeight;
      }, 600);
      document.getElementById('user-input').disabled = true;
      document.getElementById('user-input').placeholder = '';
    } else {
      addMsg(reply, 'bot');
      btn.disabled = false;
    }
  }, 800 + Math.floor(Math.random()*500));
}

function sendMsg() {
  var inp = document.getElementById('user-input');
  var text = inp.value.trim();
  if (!text) return;
  inp.value = ''; inp.style.height = '44px';
  handleSend(text);
}

function restartChat() {
  phase = 'initial';
  try { localStorage.removeItem('hold_chat'); } catch(e) {}
  document.getElementById('chat-body').innerHTML = '';
  var inp = document.getElementById('user-input');
  inp.disabled = false; inp.placeholder = '今、心にあることをそのまま…';
  document.getElementById('send-btn').disabled = false;
  addMsg(userName + 'さん、またここに戻ってきてくれてありがとう。🌿\n\n今、心にあることをそのまま話してみてください。', 'bot');
}

function renderCards() {
  var list = document.getElementById('cards-list');
  list.innerHTML = '';
  var n = 1;
  CATS.forEach(function(cat) {
    var sec = document.createElement('div');
    sec.className = 'cat-sec';
    var tt = document.createElement('div');
    tt.className = 'cat-ttl';
    tt.textContent = cat.title;
    sec.appendChild(tt);
    var grid = document.createElement('div');
    grid.className = 'cards-grid';
    cat.cards.forEach(function(c) {
      var num = n++;
      var locked = !unlocked && num > 5;
      var el = document.createElement('div');
      el.className = 'acard' + (doneCards[num] ? ' done' : '');
      el.innerHTML = '<div class="cnum">' + num + '</div><div class="cchk">✓</div><div class="ctxt">' + c.t + '</div>' + (locked ? '<div class="lock-ov">🔒</div>' : '');
      if (locked) {
        el.onclick = showModal;
      } else {
        (function(nn){ el.onclick = function(){ if(doneCards[nn]){delete doneCards[nn];}else{doneCards[nn]=true;} renderCards(); }; })(num);
      }
      grid.appendChild(el);
    });
    sec.appendChild(grid);
    list.appendChild(sec);
  });
}

function showModal() { document.getElementById('modal').classList.add('show'); }
function purchase() {
  document.getElementById('modal').classList.remove('show');
  unlocked = true;
  document.getElementById('buy-banner').style.display = 'none';
  renderCards();
}

function saveHistory() {
  try {
    var msgs = [];
    document.querySelectorAll('#chat-body .msg').forEach(function(m) {
      var b = m.querySelector('.bubble');
      if (b) msgs.push({
        type: m.classList.contains('user') ? 'user' : m.classList.contains('closing') ? 'closing' : 'bot',
        html: b.innerHTML
      });
    });
    localStorage.setItem('hold_chat', JSON.stringify({msgs:msgs, phase:phase}));
  } catch(e) {}
}

function tryLoadHistory() {
  try {
    var saved = localStorage.getItem('hold_chat');
    if (!saved) return;
    var data = JSON.parse(saved);
    if (!data.msgs || !data.msgs.length) return;
    var area = document.getElementById('chat-body');
    area.innerHTML = '';
    data.msgs.forEach(function(m) {
      var div = document.createElement('div');
      div.className = 'msg ' + m.type + (m.type==='closing' ? ' closing' : '');
      div.innerHTML = '<div class="bubble">' + m.html + '</div><div class="mtime">HOLD</div>';
      area.appendChild(div);
    });
    if (data.phase) phase = data.phase;
    if (phase === 'done') {
      document.getElementById('user-input').disabled = true;
      document.getElementById('user-input').placeholder = '';
      var gb = document.createElement('button');
      gb.className = 'btn-go'; gb.textContent = '🌿 小さな安心行動を見てみる';
      gb.onclick = function(){ switchTab('cards'); }; area.appendChild(gb);
      var rb = document.createElement('button');
      rb.className = 'btn-restart'; rb.textContent = '🌿 新しい気持ちを話す';
      rb.onclick = restartChat; area.appendChild(rb);
    } else {
      document.getElementById('send-btn').disabled = false;
    }
    area.scrollTop = 99999;
  } catch(e) {}
}
